def genfibo(n):
    fibolist = []
    x = 0
    fibolist.append(x)
    y = 1
    fibolist.append(y)
    for i in range(n-2):
        z = x + y
        fibolist.append(z)
        x = y
        y = z
    return fibolist

def checkfibo(n):
    x = 0
    y = 1
    while True:
        z = x + y
        if(n == z):
            return (0, 0)
        elif(z > n):
            return (abs(n-y), abs(n-z))
        x = y
        y = z

import random
from fibo import checkfibo, genfibo

# input

R = []
for i in range(100):
    R.append(random.randint(1, 100))

# Process

L1 = []
for n in R:
    if(checkfibo(n) == (0, 0)):
        L1.append(n)



L2 = []
F = genfibo(12)
for n in R:
    for fn in F[-7:]:
        if(n % fn == 0):
            L2.append(n)
            break

# output

print('Objective 1: ', L1)
print('Objective 2: ', L2)
